CREATE TABLE UserDetail
(
    Id INT IDENTITY(100,1) PRIMARY KEY,
    UserEmail VARCHAR(120) NOT NULL,
    Gender CHAR(1),
    Mobile BIGINT,
    FatherName VARCHAR(50) NULL,
    CreatedOn DATETIME,
    ModifiedOn DATETIME,

    CONSTRAINT FK_UserDetail_Users
    FOREIGN KEY (UserEmail)
    REFERENCES Users(Email)
);
